Form

interface Form

An interface that automatically handles validation of all FormInputs present in the inputs.

// Sample Usage
data class SignupFormState(
    val name: Name = Name(""),
    val email: Email = Email(""),
    val password: Password = Password(""),
    val submissionStatus: FormSubmissionStatus = FormSubmissionStatus.Initial
) : Form {
    override val inputs = listOf(name, email, password)
}
